Se aparecer 403 Forbidden, vá ao painel Supabase → Table Editor → atletaspro_dados → RLS Policies e certifique-se que a política permite INSERT e UPDATE para anon.

Se aparecer tabela nao encontrada, execute o SQL abaixo no SQL Editor do Supabase:
CREATE TABLE IF NOT EXISTS atletaspro_dados (
  usuario TEXT PRIMARY KEY DEFAULT 'default',
  dados JSONB,
  atualizado_em TIMESTAMPTZ DEFAULT NOW()
);

-- Habilitar RLS
ALTER TABLE atletaspro_dados ENABLE ROW LEVEL SECURITY;

-- Politica: permitir tudo para anon (acesso publico por key)
CREATE POLICY "allow_all" ON atletaspro_dados
  FOR ALL TO anon USING (true) WITH CHECK (true);

-- Inserir linha inicial
INSERT INTO atletaspro_dados (usuario, dados)
  VALUES ('default', '{}'::jsonb)
  ON CONFLICT (usuario) DO NOTHING;
